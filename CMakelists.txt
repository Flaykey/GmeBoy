cmake_minimum_required(VERSION 3.20)
project(GAMEBOY VERSION 1.0 LANGUAGES C)

file(GLOB SOURCES "src/*.c")
add_executable(GAMEBOY ${SOURCES})

target_include_directories(GAMEBOY PRIVATE include)

# Detect platform and link libraries
if(WIN32)
    # Windows MSVC
    target_link_libraries(GAMEBOY PRIVATE 
        "${CMAKE_CURRENT_SOURCE_DIR}/lib/raylib.lib"
        opengl32 gdi32 winmm
    )
elseif(UNIX)
    # Linux
    find_package(raylib REQUIRED)
    target_link_libraries(GAMEBOY PRIVATE "${CMAKE_CURRENT_SOURCE_DIR}/lib/libraylib.a"a m pthread dl X11)
endif()
